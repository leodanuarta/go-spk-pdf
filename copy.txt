package main

import (
	"fmt"
	"strings"

	"github.com/jung-kurt/gofpdf"
)

func main() {
	pdf := gofpdf.New("P", "mm", "A4", "")
	pdf.SetMargins(20, 20, 20)
	pdf.AddPage()

	// Page & column setup
	pageW, pageH := pdf.GetPageSize()
	leftM, topM, rightM, bottomM := pdf.GetMargins()
	colGap := 8.0
	colW := (pageW - leftM - rightM - colGap) / 2
	colH := pageH - topM - bottomM

	// Cursor start
	curCol := 0
	x := leftM
	y := topM
	pdf.SetXY(x, y)

	// ==== TITLE (tetap full width) ====
	pdf.SetFont("Times", "B", 14)
	pdf.CellFormat(0, 8, "PERJANJIAN KERJASAMA", "", 1, "C", false, 0, "")
	pdf.CellFormat(0, 8, "APOTEK DARING", "", 1, "C", false, 0, "")
	pdf.CellFormat(0, 8, "No. SP/XXX/LGL-PPP/XXX/2025", "", 1, "C", false, 0, "")
	pdf.Ln(6)
	pdf.SetFont("Times", "", 12)

	// ==== CONTENT ====
	content := strings.Split(fullText, "\n")

	lineH := 6.0
	for _, para := range content {
		para = strings.TrimSpace(para)
		if para == "" {
			continue
		}

		// Pasal heading → tetap dalam kolom, tapi bold
		if strings.HasPrefix(para, "PASAL") ||
			strings.ToUpper(para) == "PENDAHULUAN" ||
			strings.ToUpper(para) == "JANGKA WAKTU" ||
			strings.Contains(strings.ToUpper(para), "HAK DAN KEWAJIBAN") ||
			strings.ToUpper(para) == "KERAHASIAAN" ||
			strings.Contains(strings.ToUpper(para), "SANKSI") ||
			strings.Contains(strings.ToUpper(para), "FORCE MAJEURE") ||
			strings.Contains(strings.ToUpper(para), "PENYELESAIAN SENGKETA") ||
			strings.ToUpper(para) == "PENUTUP" {

			// cek muat atau tidak
			lineCount := pdf.SplitLines([]byte(para), colW)
			neededH := float64(len(lineCount)) * lineH
			if y+neededH > topM+colH {
				if curCol == 0 {
					curCol = 1
					x = leftM + colW + colGap
					y = topM
					pdf.SetXY(x, y)
				} else {
					pdf.AddPage()
					curCol = 0
					x = leftM
					y = topM
					pdf.SetXY(x, y)
				}
			}

			// cetak heading bold di kolom
			pdf.SetFont("Times", "B", 12)
			pdf.MultiCell(colW, lineH, para, "", "L", false)
			pdf.SetFont("Times", "", 12)
			y = pdf.GetY()
			continue
		}

		// Normal paragraph
		lineCount := pdf.SplitLines([]byte(para), colW)
		neededH := float64(len(lineCount)) * lineH
		if y+neededH > topM+colH {
			if curCol == 0 {
				curCol = 1
				x = leftM + colW + colGap
				y = topM
				pdf.SetXY(x, y)
			} else {
				pdf.AddPage()
				curCol = 0
				x = leftM
				y = topM
				pdf.SetXY(x, y)
			}
		}
		pdf.MultiCell(colW, lineH, para, "", "J", false)
		y = pdf.GetY()
	}

	// ==== SIGNATURE BLOCK (tetap full width) ====
	if curCol == 1 {
		pdf.AddPage()
	}
	pdf.Ln(10)
	pdf.CellFormat(0, 6, "Jakarta, 01 Agustus 2025", "", 1, "R", false, 0, "")
	pdf.Ln(15)

	colWsig := (pageW - leftM - rightM) / 2
	pdf.CellFormat(colWsig, 6, "PIHAK PERTAMA", "", 0, "C", false, 0, "")
	pdf.CellFormat(colWsig, 6, "PIHAK KEDUA", "", 1, "C", false, 0, "")
	pdf.Ln(25)
	pdf.CellFormat(colWsig, 6, "( Yasinta Yulian Hendrata )", "", 0, "C", false, 0, "")
	pdf.CellFormat(colWsig, 6, "( ______________________ )", "", 1, "C", false, 0, "")

	// ==== SAVE ====
	err := pdf.OutputFileAndClose("agreement_twocolumn_pasalincolumn.pdf")
	if err != nil {
		fmt.Println("Error:", err)
	}
}
